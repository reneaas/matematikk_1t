:::::::::::::{tab-item} c
$$
3(2x + 4) - 2(3x - 5)
$$

:::::{answer}
$$
22
$$
:::::


:::::{solution}
\begin{align*}
    3(2x + 4) - 2(3x - 5) &= 3 \cdot 2x + 3 \cdot 4 - 2 \cdot 3x + 2 \cdot 5 \\
    \\
    &= 6x + 12 - 6x + 10 \\
    \\
    &= 22
\end{align*}

:::::::::::::{tab-item} d
$$
3(2x + y) - 2(x - 5y) + 4(2x - y)
$$


:::::{solution}
\begin{align*}
    3(2x + y) - 2(x - 5y) + 4(2x - y) &= 3 \cdot 2x + 3 \cdot y - 2 \cdot x + 2 \cdot 5y + 4 \cdot 2x - 4 \cdot y \\
    \\
    &= 6x + 3y - 2x + 10y + 8x - 4y \\
    \\
    &= (6x - 2x + 8x) + (3y + 10y - 4y) \\
    \\
    &= 12x + 9y
\end{align*}
